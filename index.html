  <script>

        document.getElementById("letterForm").addEventListener("submit", async function(e) {

          e.preventDefault();

          const idea = document.getElementById("idea").value;

          const reason = document.getElementById("reason").value;

          const response = await fetch("/api/generate-letter", {

            method: "POST",

            headers: {

              "Content-Type": "application/json"

            },

            body: JSON.stringify({ idea, reason })

          });

          const data = await response.json();

          document.getElementById("letter").textContent = data.letter;

          document.getElementById("letterContainer").style.display = "block";

        });

  </script>
